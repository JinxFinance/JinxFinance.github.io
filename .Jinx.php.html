14     <script>
15     const { useEffect, useState } = React;
16 
17     function JinxSystem() {
18         const [showMessage, setShowMessage] = useState(false);
19 
20         useEffect(() => {
21             const startHour = 9; // Start at 9 AM
22             const endHour = 10; // End at 10 AM
23             const currentTime = new Date();
24 
25             if (currentTime.getHours() >= startHour && currentTime.getHours() < endHour) {
26                 const intervalId = setInterval(() => {
27                     setShowMessage(true);
28                     setTimeout(() => setShowMessage(false), 270);
29                 }, 270);
30 
31                 return () => clearInterval(intervalId); // Clear interval on component unmount
32             }
33         }, []);
34 
35         return showMessage ? 'Jinx Systems' : null;
36     }
